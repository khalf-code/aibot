# Persona Progression System

> Adaptive UI complexity based on user expertise and engagement

---

## Executive Summary

This specification defines a persona-based progressive unlocking system that adapts the UI complexity to match user expertise. New users see a simplified interface focused on essential actions, while power users unlock advanced features through explicit opt-in or demonstrated engagement.

### Problem Statement

The current design attempts to satisfy all users simultaneously, resulting in:

1. **Overwhelming for newcomers** â€” Too many concepts, tabs, and options presented upfront
2. **Frustrating for experts** â€” Hidden features require hunting through menus
3. **No learning path** â€” Users don't discover features progressively
4. **One-size-fits-none** â€” Neither audience is well-served

### Design Goals

1. **Frictionless Onboarding** â€” New users can create and chat with an agent in under 2 minutes
2. **Natural Discovery** â€” Features reveal themselves as users demonstrate readiness
3. **Expert Escape Hatch** â€” Power users can unlock everything immediately
4. **Persistent Preferences** â€” System remembers user's preferred complexity level
5. **Reversible** â€” Users can downgrade their level if overwhelmed

```typescript
  /** Current tier */
  tier: PersonaTier;

  /** How the tier was determined */
  source: "default" | "explicit" | "inferred";

  /** Timestamp when tier was last changed */
  changedAt: number;

  /** Features explicitly unlocked (even if tier doesn't include them) */
  unlockedFeatures: string[];

  /** Features explicitly hidden (even if tier includes them) */
  hiddenFeatures: string[];

  /** Usage statistics for inference */
  usage: PersonaUsageStats;
}

export interface PersonaUsageStats {
  /** Total number of sessions */
  sessionCount: number;

  /** Number of agents created */
  agentsCreated: number;

  /** Number of settings modified */
  settingsModified: number;

  /** Number of toolsets created */
  toolsetsCreated: number;

  /** Number of rituals created */
  ritualsCreated: number;

  /** Has viewed raw config */
  viewedRawConfig: boolean;

  /** Has used CLI */
  usedCli: boolean;

  /** First session timestamp */
  firstSession: number;

  /** Days active */
  daysActive: number;
}
```

```typescript
  hideFeature: (featureId: string) => void;

  /** Check if a feature is visible for current persona */
  isFeatureVisible: (featureId: string) => boolean;

  /** Track usage event */
  trackUsage: (event: keyof PersonaUsageStats, value?: number | boolean) => void;

  /** Check if user qualifies for tier upgrade */
  checkUpgradeEligibility: () => PersonaTier | null;

  /** Reset to default persona */
  reset: () => void;
}

const TIER_ORDER: PersonaTier[] = ["casual", "engaged", "expert"];

const DEFAULT_USAGE: PersonaUsageStats = {
  sessionCount: 0,
  agentsCreated: 0,
  settingsModified: 0,
  toolsetsCreated: 0,
  ritualsCreated: 0,
  viewedRawConfig: false,
  usedCli: false,
  firstSession: Date.now(),
  daysActive: 0,
};

const DEFAULT_PERSONA: PersonaConfig = {
  tier: "casual",
  source: "default",
  changedAt: Date.now(),
  unlockedFeatures: [],
  hiddenFeatures: [],
  usage: DEFAULT_USAGE,
};

export const usePersonaStore = create<PersonaState>()(
  persist(
    (set, get) => ({
      ...DEFAULT_PERSONA,

      upgradeTier: (tier, source = "explicit") => {
        const currentIndex = TIER_ORDER.indexOf(get().tier);
        const newIndex = TIER_ORDER.indexOf(tier);

        // Only upgrade, never downgrade via this method
        if (newIndex > currentIndex) {
          set({
            tier,
            source,
            changedAt: Date.now(),
          });
        }
      },

      downgradeTier: (tier) => {
        const currentIndex = TIER_ORDER.indexOf(get().tier);
        const newIndex = TIER_ORDER.indexOf(tier);

        // Only downgrade, never upgrade via this method
        if (newIndex < currentIndex) {
          set({
            tier,
            source: "explicit",
            changedAt: Date.now(),
          });
        }
      },

      unlockFeature: (featureId) => {
        set((state) => ({
          unlockedFeatures: [...new Set([...state.unlockedFeatures, featureId])],
        }));
      },

      hideFeature: (featureId) => {
        set((state) => ({
          hiddenFeatures: [...new Set([...state.hiddenFeatures, featureId])],
        }));
      },

      isFeatureVisible: (featureId) => {
        const state = get();

        // Explicitly hidden always wins
        if (state.hiddenFeatures.includes(featureId)) return false;

        // Explicitly unlocked always wins
        if (state.unlockedFeatures.includes(featureId)) return true;

        // Otherwise, check tier visibility matrix
        return FEATURE_VISIBILITY[state.tier]?.[featureId] ?? false;
      },

      trackUsage: (event, value) => {
        set((state) => {
          const newUsage = { ...state.usage };

          if (typeof value === "boolean") {
            (newUsage as Record<string, unknown>)[event] = value;
          } else if (typeof value === "number") {
            (newUsage as Record<string, number>)[event] = value;
          } else {
            // Increment counter
            (newUsage as Record<string, number>)[event] =
              ((newUsage as Record<string, number>)[event] || 0) + 1;
          }

          return { usage: newUsage };
        });
      },

      checkUpgradeEligibility: () => {
        const { tier, usage } = get();

        if (tier === "casual") {
          // Check for Casual â†’ Engaged signals
          const signals = [
            usage.agentsCreated >= 3,
            usage.daysActive >= 7,
            usage.settingsModified >= 1,
          ];
          if (signals.filter(Boolean).length >= 2) {
            return "engaged";
          }
        }

        if (tier === "engaged") {
          // Check for Engaged â†’ Expert signals
          const signals = [
            usage.toolsetsCreated >= 1,
            usage.settingsModified >= 10,
            usage.viewedRawConfig,
            usage.usedCli,
          ];
          if (signals.filter(Boolean).length >= 3) {
            return "expert";
          }
        }

        return null;
      },

      reset: () => set(DEFAULT_PERSONA),
    }),
    {
      name: "clawdbrain-persona",
    }
  )
);
```

### Hook: useFeatureGate

```typescript
// src/hooks/useFeatureGate.ts

import { usePersonaStore } from "@/stores/usePersonaStore";
import { useCallback, useMemo } from "react";

export interface UseFeatureGateResult {
  /** Whether the feature is visible */
  isVisible: boolean;

  /** The minimum tier required for this feature */
  requiredTier: PersonaTier;

  /** Current user tier */
  currentTier: PersonaTier;

  /** Whether user can upgrade to see this feature */
  canUpgrade: boolean;

  /** Trigger upgrade flow */
  requestUpgrade: () => void;
}

export function useFeatureGate(featureId: string): UseFeatureGateResult {
  const {
    tier,
    isFeatureVisible,
    upgradeTier,
  } = usePersonaStore();

  const isVisible = useMemo(
    () => isFeatureVisible(featureId),
    [featureId, isFeatureVisible]
  );

  const requiredTier = useMemo(() => {
    // Determine minimum tier that has this feature
    if (FEATURE_VISIBILITY.casual?.[featureId]) return "casual";
    if (FEATURE_VISIBILITY.engaged?.[featureId]) return "engaged";
    return "expert";
  }, [featureId]);

  const canUpgrade = TIER_ORDER.indexOf(requiredTier) > TIER_ORDER.indexOf(tier);

  const requestUpgrade = useCallback(() => {
    // This would trigger the upgrade confirmation dialog
    // Implementation depends on your dialog system
    window.dispatchEvent(
      new CustomEvent("persona:upgrade-requested", {
        detail: { targetTier: requiredTier, featureId },
      })
    );
  }, [requiredTier, featureId]);

  return {
    isVisible,
    requiredTier,
    currentTier: tier,
    canUpgrade,
    requestUpgrade,
  };
}
```
